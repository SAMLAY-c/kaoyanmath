```dataview
TABLE WITHOUT ID
  file.link as "笔记",
  topic as "主题",
  date as "学习日期",
  "🔴 待复习" as "状态"
FROM "02_原子笔记"
WHERE 
  date < date(this.file.name) AND
  !contains(tags, "review") AND
  (date + dur(1 days) <= date(this.file.name) OR
   date + dur(7 days) <= date(this.file.name) OR
   date + dur(30 days) <= date(this.file.name))
SORT date ASC
LIMIT 10
```
